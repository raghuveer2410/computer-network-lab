# Experiment 6 - Multiple Access Protocols

! Hub-based LAN for CSMA/CD collision demo
! PC0: IP=192.168.1.10/24
! PC1: IP=192.168.1.11/24
! PC2: IP=192.168.1.12/24
! All connected to HUB0

! Wireless LAN for CSMA/CA demo
interface Dot11Radio0
 ssid CN_Lab_Wifi
  authentication open
  guest-mode
 no shutdown
ip address 192.168.2.1 255.255.255.0

! DHCP for wireless clients
ip dhcp pool WIFI_POOL
 network 192.168.2.0 255.255.255.0
 default-router 192.168.2.1
